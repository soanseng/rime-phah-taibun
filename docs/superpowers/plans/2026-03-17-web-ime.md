# Web IME Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a browser-based Taiwanese Hokkien IME that works with zero installation, allowing users to type Taiwanese directly on a web page as a trial/demo tool and lightweight alternative to the desktop Rime-based IME.

**Architecture:** A single-page web app with a virtual IME engine in JavaScript. The core Lua logic (tone diacritics, hanlo conversion, TL↔POJ, dictionary lookup) is ported to TypeScript/JavaScript modules. Dictionary data (6MB raw, ~600KB gzipped) is pre-converted from YAML/TSV to JSON and loaded on demand. The UI provides a text input area with a candidate dropdown, output mode toggle, and copy-to-clipboard. No backend required — fully static, deployable to GitHub Pages.

**Tech Stack:** TypeScript, Vite (bundler), vanilla CSS (no framework — keep it minimal), GitHub Pages deployment

```typescript
// src/engine/tone.ts
// Converts TL numeric tones to Unicode diacritics

/**
 * Tone number → Unicode combining character mapping.
 * Tones 1 and 4 have no diacritic mark.
 */
const TONE_MARKS: Record<string, string> = {
  "2": "\u0301", // ́  acute accent
  "3": "\u0300", // ̀  grave accent
  "5": "\u0302", // ̂  circumflex
  "7": "\u0304", // ̄  macron
  "8": "\u030D", // ̍  vertical line above
  "9": "\u0306", // ̆  breve
};

/**
 * Add tone diacritic to a single TL syllable.
 * Mirrors the Lua implementation in phah_taibun_data.lua exactly.
 *
 * Placement priority:
 *   1. "oo" → mark first o
 *   2. "a", then "e" (with special "ere" handling), then "o"
 *   3. When both "i" and "u" present → mark the one appearing LATER (main vowel)
 *   4. Syllabic consonants: m, n
 *
 * @param syllable - e.g. "gua2", "tsiah8", "si1"
 * @returns syllable with Unicode combining mark, e.g. "guá", "tsia̍h", "si"
 */
export function addToneToSyllable(syllable: string): string {
  if (!syllable || syllable.length < 2) return syllable;

  const lastChar = syllable[syllable.length - 1];
  const mark = TONE_MARKS[lastChar];

  // Tones 1, 4: strip the number, no diacritic
  if (lastChar === "1" || lastChar === "4") {
    return syllable.slice(0, -1);
  }

  // Not a tone number at end: return as-is
  if (!mark) return syllable;

  const base = syllable.slice(0, -1);

  // Helper: insert combining mark after character at position
  const insertMark = (pos: number): string =>
    base.slice(0, pos) + base[pos] + mark + base.slice(pos + 1);

  // 1. Check for "oo" first (mark the first o)
  const ooIdx = base.indexOf("oo");
  if (ooIdx !== -1) return insertMark(ooIdx);

  // 2. Check "a"
  const aIdx = base.indexOf("a");
  if (aIdx !== -1) return insertMark(aIdx);

  // 3. Check "e" — special case: "ere" marks the SECOND e
  const ereIdx = base.indexOf("ere");
  if (ereIdx !== -1) return insertMark(ereIdx + 2);
  const eIdx = base.indexOf("e");
  if (eIdx !== -1) return insertMark(eIdx);

  // 4. Check "o" (single o, not oo which was handled above)
  const oIdx = base.indexOf("o");
  if (oIdx !== -1) return insertMark(oIdx);

  // 5. When both "i" and "u" present, mark the LATER one (main vowel)
  //    e.g., "hui" → mark i (pos 2); "liu" → mark u (pos 2)
  const iIdx = base.indexOf("i");
  const uIdx = base.indexOf("u");
  if (iIdx !== -1 && uIdx !== -1) return insertMark(Math.max(iIdx, uIdx));
  if (uIdx !== -1) return insertMark(uIdx);
  if (iIdx !== -1) return insertMark(iIdx);

  // 6. Syllabic consonants: m, n
  const mnIdx = base.search(/[mn]/);
  if (mnIdx !== -1) return insertMark(mnIdx);

  return base;
}

/**
 * Format a raw TL romanization string into display form with diacritics.
 *
 * Input:  "gua2 ai3 li2" (space-separated, tone numbers)
 * Output: "guá-ài-lí" (hyphen-joined, Unicode diacritics)
 *
 * Double-space indicates light tone: "tng2  lai5" → "tn̂g--lâi"
 *
 * @param roman - Raw romanization with tone numbers
 * @returns Formatted string with diacritics and hyphens
 */
export function formatRomanization(roman: string): string {
  if (!roman) return "";

  // Split on double-space (light-tone boundary)
  const groups = roman.split(/  +/);

  const formatted = groups.map((group) => {
    const syllables = group.trim().split(/\s+/);
    return syllables
      .filter((s) => s.length > 0)
      .map(addToneToSyllable)
      .join("-");
  });

  return formatted.join("--");
}
```

```typescript
// src/engine/romanization.ts
// Bidirectional TL ↔ POJ romanization conversion

/**
 * Convert TL (Tâi-lô) romanization to POJ (Pe̍h-ōe-jī).
 * Order matters: longer patterns must be replaced before shorter ones
 * to avoid partial matches (e.g., tsh before ts).
 *
 * Boundary-aware: ing/ik only convert when followed by non-alpha or end-of-string
 * (matching Lua's `gsub("ing([^a-z])", "eng%1")` + `gsub("ing$", "eng")`).
 */
export function tlToPoj(tl: string): string {
  let result = tl;
  // Order: longest match first
  result = result.replace(/tsh/g, "chh");
  result = result.replace(/ts/g, "ch");
  // Boundary-aware: only convert ing/ik at syllable boundaries
  result = result.replace(/ing(?=[^a-z]|$)/g, "eng");
  result = result.replace(/ik(?=[^a-z]|$)/g, "ek");
  result = result.replace(/ua/g, "oa");
  result = result.replace(/ue/g, "oe");
  result = result.replace(/nn/g, "\u207F"); // ⁿ superscript n
  // Handle oo with combining diacritics (when called on already-formatted text)
  // Match: o + combining mark (U+0300-U+036F) + o → o + mark + dot-above-right
  result = result.replace(/o([\u0300-\u036F])o/g, "o$1\u0358");
  // Handle plain oo (when called on raw text)
  result = result.replace(/oo/g, "o\u0358"); // o͘  combining dot above right
  return result;
}

/**
 * Fix POJ diacritics placement for oa/oe diphthongs.
 * In POJ, the tone mark for "oa" goes on the "o", not the "a".
 * Must be called AFTER formatRomanization + tlToPoj.
 *
 * Mirrors Lua's poj_fix_diacritics() in phah_taibun_data.lua.
 */
export function pojFixDiacritics(text: string): string {
  // Move combining mark from after a/e to after o in oa/oe diphthongs
  let result = text;
  result = result.replace(/oa([\u0300-\u036F])/g, "o$1a");
  result = result.replace(/oe([\u0300-\u036F])/g, "o$1e");
  return result;
}

/**
 * Convert POJ (Pe̍h-ōe-jī) romanization to TL (Tâi-lô).
 * Reverse of tlToPoj. Order matters for same reason.
 */
export function pojToTl(poj: string): string {
  let result = poj;
  // Order: longest match first
  result = result.replace(/\u207F/g, "nn"); // ⁿ → nn
  result = result.replace(/o\u0358/g, "oo"); // o͘ → oo
  result = result.replace(/chh/g, "tsh");
  result = result.replace(/ch/g, "ts");
  result = result.replace(/eng(?=[^a-z]|$)/g, "ing");
  result = result.replace(/ek(?=[^a-z]|$)/g, "ik");
  result = result.replace(/oa/g, "ua");
  result = result.replace(/oe/g, "ue");
  return result;
}
```

```typescript
// src/engine/hanlo.ts
// Hanlo (漢羅) conversion: decides whether each character outputs
// as Han character or romanization based on rules.

import { formatRomanization } from "./tone";
import type { HanloRule } from "./types";

/** Map of word/character → HanloRule */
export type HanloRules = Record<string, HanloRule>;

/**
 * Check if a word is classified as "lo" (romanization) type.
 * Unknown words default to "han" (keep as Han character).
 */
export function isLoType(word: string, rules: HanloRules): boolean {
  return rules[word]?.type === "lo";
}

/**
 * Split a string into individual Unicode characters.
 * Handles multi-byte CJK characters correctly.
 */
function splitChars(text: string): string[] {
  return [...text];
}

/**
 * Apply hanlo rules to transform candidate text.
 *
 * For each character in `text`:
 * - If its hanlo type is "lo": replace with formatted romanization
 * - If its hanlo type is "han" (or unknown): keep as Han character
 *
 * @param text - Han character text, e.g. "食飯"
 * @param rawRoman - TL romanization with tone numbers, e.g. "tsiah8 png7"
 * @param rules - Hanlo rules dictionary
 * @returns Mixed Han-Lo text, e.g. "tsia̍h飯"
 */
export function applyHanloRules(
  text: string,
  rawRoman: string,
  rules: HanloRules,
): string {
  // Check whole word first
  if (rules[text]?.type === "lo") {
    return formatRomanization(rawRoman);
  }

  const chars = splitChars(text);
  if (chars.length <= 1) return text;

  // Split on whitespace and hyphens, matching Lua's gmatch("[^%s%-]+")
  const syllables = rawRoman.split(/[\s-]+/).filter((s) => s.length > 0);
  if (syllables.length !== chars.length) return text;

  let changed = false;
  const result = chars.map((char, i) => {
    if (rules[char]?.type === "lo") {
      changed = true;
      return formatRomanization(syllables[i]);
    }
    return char;
  });

  return changed ? result.join("") : text;
}
```

### Task 6: Dictionary Data Converter (Python)

**Files:**
- Create: `scripts/convert_dict_to_json.py`

This script converts the Rime dict.yaml + hanlo_rules.yaml + hoabun_map.txt into compact JSON files for the web IME.

- [ ] **Step 1: Write the converter script**

```python
#!/usr/bin/env python3
"""Convert Rime dictionary files to JSON for the web IME.

Reads:
  - schema/phah_taibun.dict.yaml → web/public/data/dictionary.json
  - schema/hanlo_rules.yaml → web/public/data/hanlo_rules.json
  - schema/hoabun_map.txt → web/public/data/hoabun_map.json
  - schema/lighttone_rules.json → web/public/data/lighttone_rules.json (copy)

Output format optimized for size: short keys, no redundancy.
"""

import json
import re
import shutil
import sys
from pathlib import Path


def convert_dictionary(input_path: Path, output_path: Path) -> int:
    """Convert phah_taibun.dict.yaml → dictionary.json

    Output format: { "tsiah8": [["食", 1593], ["惜", 200]], ... }
    Keyed by TL romanization (hyphen-separated → space-separated).
    """
    entries: dict[str, list] = {}
    in_entries = False
    count = 0

    with open(input_path, encoding="utf-8") as f:
        for line in f:
            line = line.rstrip("\n")

            if line == "...":
                in_entries = True
                continue

            if not in_entries:
                continue

            if not line or line.startswith("#"):
                continue

            parts = line.split("\t")
            if len(parts) < 2:
                continue

            text = parts[0]
            roman = parts[1].replace("-", " ")
            weight = int(parts[2]) if len(parts) > 2 else 0

            if roman not in entries:
                entries[roman] = []
            entries[roman].append([text, weight])
            count += 1

    # Sort each entry list by weight descending
    for key in entries:
        entries[key].sort(key=lambda x: -x[1])

    output_path.parent.mkdir(parents=True, exist_ok=True)
    with open(output_path, "w", encoding="utf-8") as f:
        json.dump(entries, f, ensure_ascii=False, separators=(",", ":"))

    return count


def convert_hanlo_rules(input_path: Path, output_path: Path) -> int:
    """Convert hanlo_rules.yaml → hanlo_rules.json

    Output format: { "食": "lo", "飯": "han", ... }
    Simplified to just word → type mapping.
    """
    rules: dict[str, str] = {}

    with open(input_path, encoding="utf-8") as f:
        current_word = None
        for line in f:
            line = line.rstrip("\n")

            if not line or line.startswith("#"):
                continue

            # Top-level key (word)
            if not line.startswith(" ") and line.endswith(":"):
                current_word = line[:-1].strip('"').strip("'")
                continue

            # Nested key
            if current_word and "type:" in line:
                match = re.search(r"type:\s*(\w+)", line)
                if match:
                    rules[current_word] = match.group(1)

    output_path.parent.mkdir(parents=True, exist_ok=True)
    with open(output_path, "w", encoding="utf-8") as f:
        json.dump(rules, f, ensure_ascii=False, separators=(",", ":"))

    return len(rules)


def convert_hoabun_map(input_path: Path, output_path: Path) -> int:
    """Convert hoabun_map.txt → hoabun_map.json

    Output format: { "吃": "tsiah8", "好": "ho2", ... }
    """
    mapping: dict[str, str] = {}

    with open(input_path, encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line or line.startswith("#"):
                continue
            parts = line.split("\t")
            if len(parts) >= 2:
                mapping[parts[0]] = parts[1]

    output_path.parent.mkdir(parents=True, exist_ok=True)
    with open(output_path, "w", encoding="utf-8") as f:
        json.dump(mapping, f, ensure_ascii=False, separators=(",", ":"))

    return len(mapping)


def main():
    project_root = Path(__file__).parent.parent
    schema_dir = project_root / "schema"
    output_dir = project_root / "web" / "public" / "data"

    print("Converting dictionary files for web IME...")

    count = convert_dictionary(
        schema_dir / "phah_taibun.dict.yaml",
        output_dir / "dictionary.json",
    )
    print(f"  dictionary.json: {count} entries")

    count = convert_hanlo_rules(
        schema_dir / "hanlo_rules.yaml",
        output_dir / "hanlo_rules.json",
    )
    print(f"  hanlo_rules.json: {count} rules")

    count = convert_hoabun_map(
        schema_dir / "hoabun_map.txt",
        output_dir / "hoabun_map.json",
    )
    print(f"  hoabun_map.json: {count} mappings")

    # Copy lighttone_rules.json as-is
    shutil.copy2(
        schema_dir / "lighttone_rules.json",
        output_dir / "lighttone_rules.json",
    )
    print("  lighttone_rules.json: copied")

    # Print sizes
    print("\nFile sizes:")
    for f in output_dir.glob("*.json"):
        size_kb = f.stat().st_size / 1024
        print(f"  {f.name}: {size_kb:.0f} KB")

    print("\nDone! Files written to web/public/data/")


if __name__ == "__main__":
    main()
```

```typescript
// src/engine/dictionary.ts
// Dictionary loader and lookup for the web IME

import type { Candidate } from "./types";
import { formatRomanization } from "./tone";

/** Raw dictionary data format: { "roman": [["text", weight], ...] } */
type RawDictData = Record<string, [string, number][]>;

export class Dictionary {
  private data: RawDictData = {};
  private prefixIndex: Map<string, string[]> = new Map();
  private loaded = false;

  /**
   * Load dictionary from a raw object (for testing or pre-loaded data).
   */
  loadFromObject(data: RawDictData): void {
    this.data = data;
    this.buildPrefixIndex();
    this.loaded = true;
  }

  /**
   * Load dictionary from a URL (fetches JSON).
   */
  async loadFromUrl(url: string): Promise<void> {
    const response = await fetch(url);
    this.data = await response.json();
    this.buildPrefixIndex();
    this.loaded = true;
  }

  /**
   * Look up exact romanization match.
   * @param roman - TL romanization, e.g. "tsiah8" or "tsiah8 png7"
   * @returns Sorted candidates (highest weight first)
   */
  lookup(roman: string): Candidate[] {
    const entries = this.data[roman];
    if (!entries) return [];

    return entries.map(([text, weight]) => ({
      text,
      roman,
      comment: formatRomanization(roman),
      weight,
    }));
  }

  /**
   * Look up all phrases that start with the given completed syllables.
   * Only does exact prefix matching on completed syllable sequences
   * (not character-level prefix search, which would be O(n) over 200K entries).
   *
   * @param completedSyllables - Space-separated completed syllables, e.g. "tsiah8 png7"
   * @returns Candidates from exact match + longer phrases starting with these syllables
   */
  phraseLookup(completedSyllables: string): Candidate[] {
    const results: Candidate[] = [];

    // Exact match (highest priority)
    const exact = this.data[completedSyllables];
    if (exact) {
      for (const [text, weight] of exact) {
        results.push({
          text,
          roman: completedSyllables,
          comment: formatRomanization(completedSyllables),
          weight: weight + 1000, // Boost exact matches
        });
      }
    }

    // Phrase extensions: find longer phrases starting with these syllables
    // Uses cached prefix index for O(1) lookup instead of O(n) scan
    const extensions = this.prefixIndex.get(completedSyllables);
    if (extensions) {
      for (const roman of extensions) {
        if (roman === completedSyllables) continue; // Already added
        const entries = this.data[roman];
        if (entries) {
          for (const [text, weight] of entries) {
            results.push({
              text,
              roman,
              comment: formatRomanization(roman),
              weight,
            });
          }
        }
      }
    }

    results.sort((a, b) => b.weight - a.weight);
    return results;
  }

  /**
   * Build prefix index at load time for fast phrase lookup.
   * Maps first-syllable → list of full romanization keys.
   * Called automatically after loading data.
   */
  private buildPrefixIndex(): void {
    this.prefixIndex = new Map();
    for (const roman of Object.keys(this.data)) {
      // Index by first syllable, first two syllables, etc.
      const syllables = roman.split(" ");
      for (let i = 1; i <= syllables.length; i++) {
        const prefix = syllables.slice(0, i).join(" ");
        if (!this.prefixIndex.has(prefix)) {
          this.prefixIndex.set(prefix, []);
        }
        this.prefixIndex.get(prefix)!.push(roman);
      }
    }
  }

  get isLoaded(): boolean {
    return this.loaded;
  }
}
```

```typescript
// src/engine/ime-engine.ts
// Core IME engine: manages input state, candidate generation, and output

import { Dictionary } from "./dictionary";
import { applyHanloRules, type HanloRules } from "./hanlo";
import { formatRomanization } from "./tone";
import { tlToPoj, pojFixDiacritics } from "./romanization";
import type { Candidate, EngineState } from "./types";
import { OutputMode } from "./types";

export class ImeEngine {
  readonly dictionary: Dictionary;
  private hanloRules: HanloRules = {};
  private _state: EngineState;

  /** Callback fired whenever state changes */
  onStateChange?: (state: EngineState) => void;

  constructor() {
    this.dictionary = new Dictionary();
    this._state = {
      inputBuffer: "",
      outputMode: OutputMode.HanLoTL,
      candidates: [],
      selectedIndex: 0,
      committedText: "",
    };
  }

  get state(): Readonly<EngineState> {
    return this._state;
  }

  loadHanloRules(rules: HanloRules): void {
    this.hanloRules = rules;
  }

  /**
   * Process romanization input and update candidates.
   */
  processInput(input: string): void {
    this._state.inputBuffer = input;
    this._state.selectedIndex = 0;

    // Look up candidates using syllable-based phrase lookup
    const raw = this.dictionary.phraseLookup(input);

    // Apply output mode transformations
    this._state.candidates = raw.map((c) => this.transformCandidate(c));
    this.notify();
  }

  /**
   * Select and commit a candidate by index.
   */
  selectCandidate(index: number): void {
    const candidate = this._state.candidates[index];
    if (!candidate) return;

    const outputMode = this._state.outputMode;
    let output: string;

    if (outputMode === OutputMode.FullTL || outputMode === OutputMode.FullPOJ) {
      // Full romanization mode: output romanization
      let roman = formatRomanization(candidate.roman);
      if (outputMode === OutputMode.FullPOJ) {
        roman = pojFixDiacritics(tlToPoj(roman));
      }
      output = roman;
    } else {
      // Han-Lo mode: apply hanlo rules
      output = applyHanloRules(candidate.text, candidate.roman, this.hanloRules);
      if (outputMode === OutputMode.HanLoPOJ) {
        output = pojFixDiacritics(tlToPoj(output));
      }
    }

    this._state.committedText += output;
    this._state.inputBuffer = "";
    this._state.candidates = [];
    this._state.selectedIndex = 0;
    this.notify();
  }

  /**
   * Clear current input without committing.
   */
  clear(): void {
    this._state.inputBuffer = "";
    this._state.candidates = [];
    this._state.selectedIndex = 0;
    this.notify();
  }

  /**
   * Set the output mode.
   */
  setOutputMode(mode: OutputMode): void {
    this._state.outputMode = mode;
    // Re-process current input if any
    if (this._state.inputBuffer) {
      this.processInput(this._state.inputBuffer);
    }
    this.notify();
  }

  /**
   * Move candidate selection up/down.
   */
  moveSelection(delta: number): void {
    const len = this._state.candidates.length;
    if (len === 0) return;
    this._state.selectedIndex =
      (this._state.selectedIndex + delta + len) % len;
    this.notify();
  }

  /**
   * Reset all state (for "new document" or full clear).
   */
  reset(): void {
    this._state.inputBuffer = "";
    this._state.candidates = [];
    this._state.selectedIndex = 0;
    this._state.committedText = "";
    this.notify();
  }

  private transformCandidate(c: Candidate): Candidate {
    const mode = this._state.outputMode;
    let comment = formatRomanization(c.roman);

    if (mode === OutputMode.HanLoPOJ || mode === OutputMode.FullPOJ) {
      comment = `TL:${comment} POJ:${tlToPoj(comment)}`;
    }

    return { ...c, comment };
  }

  private notify(): void {
    this.onStateChange?.(this._state);
  }
}
```

```typescript
// src/ui/toolbar.ts
// Output mode toggle toolbar

import { OutputMode } from "../engine/types";

const MODE_LABELS: Record<OutputMode, string> = {
  [OutputMode.HanLoTL]: "漢羅 TL",
  [OutputMode.HanLoPOJ]: "漢羅 POJ",
  [OutputMode.FullTL]: "全羅 TL",
  [OutputMode.FullPOJ]: "全羅 POJ",
};

export function createToolbar(
  container: HTMLElement,
  currentMode: OutputMode,
  onModeChange: (mode: OutputMode) => void,
): { setMode: (mode: OutputMode) => void } {
  const buttons: HTMLButtonElement[] = [];

  for (const [modeStr, label] of Object.entries(MODE_LABELS)) {
    const mode = Number(modeStr) as OutputMode;
    const btn = document.createElement("button");
    btn.textContent = label;
    btn.dataset.mode = modeStr;

    if (mode === currentMode) {
      btn.classList.add("active");
    }

    btn.addEventListener("click", () => {
      onModeChange(mode);
    });

    buttons.push(btn);
    container.appendChild(btn);
  }

  return {
    setMode(mode: OutputMode) {
      for (const btn of buttons) {
        btn.classList.toggle("active", Number(btn.dataset.mode) === mode);
      }
    },
  };
}
```

```typescript
// src/ui/input-panel.ts
// Handles IME input field and candidate dropdown

import type { Candidate } from "../engine/types";

/**
 * Two-phase input model:
 * 1. COMPOSING: User is typing romanization. Number keys 1-9 are tone numbers.
 *    Space acts as syllable separator.
 * 2. SELECTING: A complete syllable is detected (input ends with tone number 1-9).
 *    Candidates are shown. Space selects first candidate. Number keys select by index.
 *
 * A syllable is "complete" when the input buffer ends with a tone number [1-9].
 * This matches how Rime's speller works.
 */
export interface InputPanelCallbacks {
  onInput: (value: string) => void;
  onSelect: (index: number) => void;
  onClear: () => void;
  onMoveSelection: (delta: number) => void;
}

export function createInputPanel(
  inputEl: HTMLInputElement,
  candidateListEl: HTMLElement,
  callbacks: InputPanelCallbacks,
): {
  updateCandidates: (candidates: Candidate[], selectedIndex: number) => void;
  clearInput: () => void;
} {
  // Track composing state for multi-byte input methods
  let isComposing = false;

  inputEl.addEventListener("compositionstart", () => {
    isComposing = true;
  });

  inputEl.addEventListener("compositionend", () => {
    isComposing = false;
    callbacks.onInput(inputEl.value);
  });

  inputEl.addEventListener("input", () => {
    if (!isComposing) {
      callbacks.onInput(inputEl.value);
    }
  });

  /**
   * Determine if candidates are visible and input ends with a complete syllable
   * (tone number 1-9). Only then do space/number keys act as selection keys.
   */
  function isInSelectionPhase(): boolean {
    if (candidateListEl.children.length === 0) return false;
    const value = inputEl.value;
    // A syllable is complete when input ends with a tone number
    return /[1-9]$/.test(value);
  }

  inputEl.addEventListener("keydown", (e) => {
    if (isComposing) return;

    switch (e.key) {
      case "ArrowDown":
        e.preventDefault();
        callbacks.onMoveSelection(1);
        break;

      case "ArrowUp":
        e.preventDefault();
        callbacks.onMoveSelection(-1);
        break;

      case "Escape":
        e.preventDefault();
        callbacks.onClear();
        break;

      case "Enter":
        // Enter always selects the current candidate if in selection phase
        if (isInSelectionPhase()) {
          e.preventDefault();
          callbacks.onSelect(0);
        }
        break;

      case " ":
        // Space selects first candidate ONLY in selection phase
        // Otherwise it's a syllable separator (passed through to input)
        if (isInSelectionPhase()) {
          e.preventDefault();
          callbacks.onSelect(0);
        }
        // If not in selection phase, space goes into input buffer normally
        break;

      default:
        // Number keys 1-9: in selection phase, select candidate
        // Otherwise, they're tone numbers (passed through to input)
        // Note: this means the user must type the tone number to complete
        // a syllable before they can use number keys for selection.
        // To select the 2nd candidate for "gua2", user types "gua2" then
        // presses "2" again (first "2" was the tone, second "2" selects).
        break;
    }
  });

  function updateCandidates(candidates: Candidate[], selectedIndex: number) {
    candidateListEl.innerHTML = "";

    if (candidates.length === 0) {
      candidateListEl.hidden = true;
      return;
    }

    candidateListEl.hidden = false;

    candidates.slice(0, 9).forEach((candidate, i) => {
      const item = document.createElement("div");
      item.className = "candidate-item";
      if (i === selectedIndex) item.classList.add("selected");

      // Use textContent instead of innerHTML to prevent XSS
      const indexSpan = document.createElement("span");
      indexSpan.className = "candidate-index";
      indexSpan.textContent = String(i + 1);

      const textSpan = document.createElement("span");
      textSpan.className = "candidate-text";
      textSpan.textContent = candidate.text;

      const commentSpan = document.createElement("span");
      commentSpan.className = "candidate-comment";
      commentSpan.textContent = candidate.comment;

      item.appendChild(indexSpan);
      item.appendChild(textSpan);
      item.appendChild(commentSpan);

      item.addEventListener("click", () => callbacks.onSelect(i));
      candidateListEl.appendChild(item);
    });
  }

  function clearInput() {
    inputEl.value = "";
    candidateListEl.innerHTML = "";
    candidateListEl.hidden = true;
  }

  return { updateCandidates, clearInput };
}
```

```typescript
// src/main.ts
// App entry point: connects IME engine to UI components

import { ImeEngine } from "./engine/ime-engine";
import { OutputMode } from "./engine/types";
import { createToolbar } from "./ui/toolbar";
import { createInputPanel } from "./ui/input-panel";
import type { HanloRules } from "./engine/hanlo";

async function main() {
  const engine = new ImeEngine();

  // --- Load data ---
  const outputArea = document.getElementById("output-area") as HTMLTextAreaElement;
  outputArea.placeholder = "載入字典中...";

  const [dictData, hanloData] = await Promise.all([
    fetch("data/dictionary.json").then((r) => r.json()),
    fetch("data/hanlo_rules.json").then((r) => r.json()),
  ]);

  engine.dictionary.loadFromObject(dictData);
  engine.loadHanloRules(hanloData as HanloRules);

  outputArea.placeholder = "台文會出現在這裡...";

  // --- Setup Toolbar ---
  const toolbarEl = document.getElementById("toolbar")!;
  const toolbar = createToolbar(toolbarEl, OutputMode.HanLoTL, (mode) => {
    engine.setOutputMode(mode);
    toolbar.setMode(mode);
  });

  // --- Setup Input Panel ---
  const inputEl = document.getElementById("ime-input") as HTMLInputElement;
  const candidateListEl = document.getElementById("candidate-list")!;

  const inputPanel = createInputPanel(inputEl, candidateListEl, {
    onInput(value) {
      if (value.trim()) {
        engine.processInput(value.trim());
      } else {
        engine.clear();
      }
    },
    onSelect(index) {
      engine.selectCandidate(index);
      inputPanel.clearInput();
      inputEl.focus();
    },
    onClear() {
      engine.clear();
      inputPanel.clearInput();
    },
    onMoveSelection(delta) {
      engine.moveSelection(delta);
    },
  });

  // --- State change handler ---
  engine.onStateChange = (state) => {
    inputPanel.updateCandidates(state.candidates, state.selectedIndex);
    outputArea.value = state.committedText;
  };

  // --- Copy & Clear buttons ---
  document.getElementById("copy-btn")?.addEventListener("click", () => {
    navigator.clipboard.writeText(outputArea.value);
  });

  document.getElementById("clear-btn")?.addEventListener("click", () => {
    engine.reset();
    inputPanel.clearInput();
  });
}

main();
```
